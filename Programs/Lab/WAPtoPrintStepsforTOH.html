// Tower of Hanoi.

#include <iostream>
using namespace std;

void TOH(int n, char S, char A, char D)
{
	if (n == 1)
		cout << S << "->" << D << endl;
	else
	{
		TOH(n - 1, S, D, A);
		cout << S << "->" << D << endl;
		TOH(n - 1, A, S, D);
	}
}

int main()
{
	system("CLS");
	int n;
	cout << "Enter the number of Discs: ";
	cin >> n;

	TOH(n, 'S', 'A', 'D');
	return 0;
}
