// WAP to convert Infix expression to Postfix Expression.

#include <iostream>
#include <stdlib.h>
#include <cstring>
using namespace std;

class Stack
{
	char a[100];
	int top = -1;

public:
	void push(char x)
	{
		if (top == 99)
		{
			cout << "Stack Full";
		}
		else
		{
			top++;
			a[top] = x;
		}
	}

	void pop()
	{
		if (top == -1)
		{
			cout << "Stack Empty";
		}
		else
		{
			cout << a[top];
			top--;
		}
	}

	char TOS()
	{
		if (top == -1)
		{
			return NULL;
		}
		else
		{
			return a[top];
		}
	}

	void remove_top()
	{
		top--;
	}
};

bool isoperator(char ch)
{
	char Operator[] = {"^$/*+-()"};

	for (int i = 0; i < strlen(Operator); i++)
	{
		if (ch == Operator[i])
		{
			return true;
		}
	}

	return false;
}

int precedence(char op)
{
	char o3[] = {"^$"};
	char o2[] = {"/*"};
	char o1[] = {"+-"};

	for (int i = 0; i < strlen(o3); i++)
	{
		if (op == o3[i])
		{
			return 3;
		}
	}

	for (int i = 0; i < strlen(o2); i++)
	{
		if (op == o2[i])
		{
			return 2;
		}
	}

	if (op == '(')
	{
		return 0;
	}

	if (op == NULL)
	{
		return -1;
	}

	return 1;
}

char associativity(char op)
{
	char R_L[] = {"^$"};

	for (int i = 0; i < strlen(R_L); i++)
	{
		if (op == R_L[i])
		{
			return 'R';
		}
	}

	return 'L';
}

void operator_handler(char op, Stack &s)
{
	if (op == '(')
	{
		s.push(op);
	}
	else if (op == ')')
	{
		while (s.TOS() != '(')
		{
			s.pop();
		}

		s.remove_top();
	}
	else
	{
		if (precedence(op) > precedence(s.TOS()))
		{
			s.push(op);
		}
		else if (precedence(op) < precedence(s.TOS()))
		{
			s.pop();
			operator_handler(op, s);
		}
		else
		{
			if (associativity(op) == 'R')
			{
				s.push(op);
			}
			else
			{
				s.pop();
				s.push(op);
			}
		}
	}
}

int main()
{
	system("CLS");

	Stack s;
	char *expression;
	int len;

	cout << "Enter the Infix Expression: ";
	cin >> expression;

	cout << "\nThe Postfix Expression is: ";

	len = strlen(expression);

	for (int i = 0; i < len; i++)
	{
		if (isoperator(expression[i]))
		{
			operator_handler(expression[i], s);
		}
		else
		{
			cout << expression[i];
		}
	}

	while (s.TOS() != NULL)
	{
		s.pop();
	}
	return 0;
}
// WAP to convert Infix expression to Postfix Expression - (using Stack as ADT).

#include <iostream>
#include <stdlib.h>
#include <stack>
#include <cstring>
using namespace std;

bool isoperator(char ch)
{
	if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z'))
		return false;
	return true;
}

int precedence(char op)
{
	switch (op)
	{
	case '^':
	case '$':
		return 3;

	case '/':
	case '*':
		return 2;

	case '(':
		return 0;

	default:
		return 1;
	}
}

char associativity(char op)
{
	if (op == '^' || op == '$')
		return 'R';
	else
		return 'L';
}

void operator_handler(char op, stack<char> &s)
{
	if (s.empty())
		s.push(op);
	else if (op == '(')
		s.push(op);
	else if (op == ')')
	{
		while (s.top() != '(')
		{
			cout << s.top();
			s.pop();
		}
		s.pop();
	}
	else
	{
		if (precedence(op) > precedence(s.top()))
			s.push(op);
		else if (precedence(op) < precedence(s.top()))
		{
			cout << s.top();
			s.pop();
			operator_handler(op, s);
		}
		else
		{
			if (associativity(op) == 'R')
				s.push(op);
			else
			{
				cout << s.top();
				s.pop();
				s.push(op);
			}
		}
	}
}

int main()
{
	system("CLS");

	stack<char> s;
	char expression[100], e;

	cout << "Enter the Infix Expression: ";
	cin >> expression;

	cout << "\nThe Postfix Expression is: ";

	for (int i = 0; i < strlen(expression); i++)
	{
		e = expression[i];
		if (isoperator(e))
			operator_handler(e, s);
		else
			cout << e;
	}

	while (!s.empty())
	{
		cout << s.top();
		s.pop();
	}
	return 0;
}
