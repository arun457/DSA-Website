// WAP to implement Singly Linked List using Structure.

#include <iostream>
using namespace std;

struct Node
{
	int data;
	Node *next = NULL;
};

Node *head = NULL, *temp, *previous, *newnode;

void createnode()
{
	int value;
	cout << "Enter the value to be inserted: ";
	cin >> value;

	newnode = new Node;
	newnode->data = value;
}

void insert_at_beg()
{
	createnode();

	if (head == NULL)
	{
		head = newnode;
	}
	else
	{
		newnode->next = head;
		head = newnode;
	}
}

void insert_at_end()
{
	createnode();
	if (head == NULL)
	{
		head = newnode;
	}
	else
	{
		temp = head;
		while (temp->next != NULL)
		{
			temp = temp->next;
		}
		temp->next = newnode;
	}
}

void insert_at_pos()
{
	createnode();

	int pos;
	cout << "Enter the position: ";
	cin >> pos;

	if (head == NULL)
	{
		if (pos == 1)
		{
			head = newnode;
		}
		else
		{
			cout << "Linked List Empty!!" << endl;
		}
	}
	else
	{
		temp = head;
		for (int i = 1; i < pos - 1; i++)
		{
			temp = temp->next;

			if (temp == NULL)
			{
				cout << "The required position doesnot exist." << endl;
				return;
			}
		}

		newnode->next = temp->next;
		temp->next = newnode;
	}
}

void delete_from_beg()
{
	if (head == NULL)
	{
		cout << "Linked List Empty" << endl;
	}
	else
	{
		temp = head;
		head = temp->next;
		delete temp;
	}
}

void delete_from_end()
{
	if (head == NULL)
	{
		cout << "Linked List Empty" << endl;
	}
	else
	{
		temp = head;
		while (temp->next != NULL)
		{
			previous = temp;
			temp = temp->next;
		}

		if (head == temp)
		{
			head = NULL;
		}
		else
		{
			previous->next = NULL;
		}
		delete temp;
	}
}

void delete_from_pos()
{
	int pos;
	cout << "Enter the position: ";
	cin >> pos;

	if (head == NULL)
	{
		cout << "Linked List Empty" << endl;
	}
	else
	{
		temp = head;
		for (int i = 1; i < pos; i++)
		{
			previous = temp;
			temp = temp->next;

			if (temp == NULL)
			{
				cout << "The required position doesnot exist." << endl;
				return;
			}
		}

		if (temp == head)
		{
			head = NULL;
		}
		else
		{
			previous->next = temp->next;
		}
		delete temp;
	}
}

void display()
{
	temp = head;
	if (head == NULL)
	{
		cout << "Linked List Empty" << endl;
	}
	else
	{
		while (temp != NULL)
		{
			cout << temp->data << endl;
			temp = temp->next;
		}
	}
}

int main()
{
	system("CLS");
	insert_at_end();
	insert_at_beg();
	delete_from_end();

	display();
	return 0;
}
