#include <iostream>
#include <stdlib.h>
#include <cstring>
using namespace std;

class Stack
{
	char a[100];
	int top = -1;

public:
	void push(char ch)
	{
		if (top == 99)
		{
			cout << "Stack Full";
		}
		else
		{
			top++;
			a[top] = ch;
		}
	}

	void pop()
	{
		if (top == -1)
		{
			cout << "Stack Empty";
		}
		else
		{
			cout << a[top];
			top--;
		}
	}

	char TOS()
	{
		if (top == -1)
		{
			return NULL;
		}
		else
		{
			return a[top];
		}
	}

	void remove_top()
	{
		top--;
	}
};

Stack s, storage;

bool isoperator(char ch)
{
	char Operator[] = {"^$/*+-()"};

	for (int i = 0; i < strlen(Operator); i++)
	{
		if (ch == Operator[i])
		{
			return true;
		}
	}

	return false;
}

int precedence(char op)
{
	char o3[] = {"^$"};
	char o2[] = {"/*"};
	char o1[] = {"+-"};

	for (int i = 0; i < strlen(o3); i++)
	{
		if (op == o3[i])
		{
			return 3;
		}
	}

	for (int i = 0; i < strlen(o2); i++)
	{
		if (op == o2[i])
		{
			return 2;
		}
	}

	if (op == '(')
	{
		return 0;
	}

	if (op == NULL)
	{
		return -1;
	}

	return 1;
}

char associativity(char op)
{
	char R_L[] = {"^$"};

	for (int i = 0; i < strlen(R_L); i++)
	{
		if (op == R_L[i])
		{
			return 'R';
		}
	}

	return 'L';
}

void operator_handler(char op)
{
	if (op == ')')
	{
		s.push(op);
	}
	else if (op == '(')
	{
		while (s.TOS() != ')')
		{
			storage.push(s.TOS());
			s.remove_top();
		}

		s.remove_top();
	}
	else
	{
		if (precedence(op) > precedence(s.TOS()))
		{
			s.push(op);
		}
		else if (precedence(op) < precedence(s.TOS()))
		{
			storage.push(s.TOS());
			s.remove_top();
			operator_handler(op);
		}
		else
		{
			if (associativity(op) == 'L')
			{
				s.push(op);
			}
			else
			{
				storage.push(s.TOS());
				s.remove_top();
				operator_handler(op);
			}
		}
	}
}

int main()
{
	system("CLS");

	char *expression;
	int len;

	cout << "Enter the Infix Expression: ";
	cin >> expression;
	expression = strrev(expression);

	cout << "\nThe Pretfix Expression is: ";

	len = strlen(expression);

	for (int i = 0; i < len; i++)
	{
		if (isoperator(expression[i]))
		{
			operator_handler(expression[i]);
		}
		else
		{
			storage.push(expression[i]);
		}
	}

	while (s.TOS() != NULL)
	{
		storage.push(s.TOS());
		s.remove_top();
	}

	while (storage.TOS() != NULL)
	{
		storage.pop();
	}

	return 0;
}
