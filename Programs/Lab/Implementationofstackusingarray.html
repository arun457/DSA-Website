// Implementation of stack using array

#include <iostream>
using namespace std;

template <class T, int Size>
class Stack
{
	T a[Size];
	int top = -1;

public:
	void push(T x)
	{
		if (top == (Size - 1))
		{
			cout << "Stack Full";
		}
		else
		{
			top++;
			a[top] = x;
		}
	}

	void pop()
	{
		if (top == -1)
		{
			cout << "Stack Empty";
		}
		else
		{
			cout << a[top] << endl;
			top--;
		}
	}

	void TOS()
	{
		if (top == -1)
		{
			cout << "Stack Empty";
		}
		else
		{
			cout << a[top] << endl;
		}
	}
};

int main()
{
	system("CLS");

	Stack<int, 5> s;
	s.push(1);
	s.push(2);
	s.push(3);
	s.push(4);
	s.push(5);

	s.TOS();

	s.pop();
	s.pop();
	s.pop();
	s.pop();
	s.pop();

	s.TOS();

	return 0;
}
