// WAP to convert Infix expression to Prefix Expression. (With Each Steps Shown)

#include <iostream>
#include <iomanip>
#include <stdlib.h>
#include <cstring>
#include <algorithm>
using namespace std;

class Stack
{
	char a[100];
	char temp[100];
	int top = -1;

public:
	void push(char ch)
	{
		if (top == 99)
		{
			cout << "Stack Full";
		}
		else
		{
			top++;
			a[top] = ch;
		}
	}

	void pop()
	{
		if (top == -1)
		{
			cout << "Stack Empty";
		}
		else
		{
			cout << a[top];
			top--;
		}
	}

	char TOS()
	{
		if (top == -1)
		{
			return NULL;
		}
		else
		{
			return a[top];
		}
	}

	void remove_top()
	{
		top--;
	}

	char *show()
	{
		std::fill(temp, (temp + 99), 0);

		for (int i = 0; i <= top; i++)
		{
			temp[i] = a[i];
		}

		return temp;
	}
};

bool isoperator(char ch)
{
	char Operator[] = {"^$/*+-()"};

	for (int i = 0; i < strlen(Operator); i++)
	{
		if (ch == Operator[i])
		{
			return true;
		}
	}

	return false;
}

int precedence(char op)
{
	char o3[] = {"^$"};
	char o2[] = {"/*"};
	char o1[] = {"+-"};

	for (int i = 0; i < strlen(o3); i++)
	{
		if (op == o3[i])
		{
			return 3;
		}
	}

	for (int i = 0; i < strlen(o2); i++)
	{
		if (op == o2[i])
		{
			return 2;
		}
	}

	if (op == ')')
	{
		return 0;
	}

	if (op == NULL)
	{
		return -1;
	}

	return 1;
}

char associativity(char op)
{
	char R_L[] = {"^$"};

	for (int i = 0; i < strlen(R_L); i++)
	{
		if (op == R_L[i])
		{
			return 'R';
		}
	}

	return 'L';
}

void operator_handler(char op, Stack &s, Stack &storage)
{
	if (op == ')')
	{
		s.push(op);
	}
	else if (op == '(')
	{
		while (s.TOS() != ')')
		{
			storage.push(s.TOS());
			s.remove_top();
		}

		s.remove_top();
	}
	else
	{
		if (precedence(op) > precedence(s.TOS()))
		{
			s.push(op);
		}
		else if (precedence(op) < precedence(s.TOS()))
		{
			storage.push(s.TOS());
			s.remove_top();
			operator_handler(op, s, storage);
		}
		else
		{
			if (associativity(op) == 'L')
			{
				s.push(op);
			}
			else
			{
				storage.push(s.TOS());
				s.remove_top();
				s.push(op);
			}
		}
	}
}
