// WAP to implement Floyd Warshall Algorithm.

#include <iostream>
#define v 4
#define inf 99999
using namespace std;

int main()
{
	system("CLS");
	int d[v][v] = {{0, 9, -4, inf},
				   {6, 0, inf, 2},
				   {inf, 5, 0, inf},
				   {inf, inf, 1, 0}},
		dist;

	for (int k = 0; k < v; k++)
	{
		for (int i = 0; i < v; i++)
		{
			for (int j = 0; j < v; j++)
			{
				dist = d[i][k] + d[k][j];
				if (dist < d[i][j])
				{
					d[i][j] = dist;
				}
			}
		}
	}

	cout << "The final distance matrix is: " << endl;
	for (int i = 0; i < v; i++)
	{
		for (int j = 0; j < v; j++)
		{
			if (d[i][j] != inf)
			{
				cout << d[i][j] << "\t";
			}
			else
			{
				cout << "Inf"
					 << "\t";
			}
		}
		cout << endl;
	}
	return 0;
}
