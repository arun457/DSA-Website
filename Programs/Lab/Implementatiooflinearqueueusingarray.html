// Implementatio of linear queue using array

#include <iostream>
using namespace std;

template <class T, int Size>
class Queue
{
	T a[Size];
	int f = -1, r = -1;

public:
	void enqueue(T x)
	{
		if (r == (Size - 1))
		{
			cout << "Queue Full";
		}
		else if (r == -1 && f == -1)
		{
			r = f = 0;
			a[r] = x;
		}
		else
		{
			r++;
			a[r] = x;
		}
	}

	void dequeue()
	{
		if (f == -1 && r == -1)
		{
			cout << "Queue Empty";
		}
		else if (f == r)
		{
			cout << a[f];
			f = r = -1;
		}
		else
		{
			cout << a[f] << endl;
			f++;
		}
	}

	void peek()
	{
		if (r == -1 && f == -1)
		{
			cout << "Queue Empty";
		}
		else
		{
			cout << a[f] << endl;
		}
	}
};

int main()
{
	system("CLS");

	Queue<int, 5> q;
	q.enqueue(1);
	q.enqueue(2);
	q.enqueue(3);
	q.enqueue(4);
	q.enqueue(5);

	q.peek();

	q.dequeue();
	q.dequeue();
	q.dequeue();
	q.dequeue();
	q.dequeue();

	return 0;
}
